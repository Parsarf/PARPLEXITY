**Step 1 — Build author format helpers:**

```python
def _apa_authors(names: list[str]) -> str:
    """Format authors for APA: 'Last, F. I., Last, F. I., & Last, F. I.'"""
    if not names:
        return ""
    formatted = []
    for name in names:
        parts = name.strip().split()
        if len(parts) >= 2:
            last = parts[-1]
            initials = " ".join(p[0] + "." for p in parts[:-1])
            formatted.append(f"{last}, {initials}")
        elif len(parts) == 1:
            formatted.append(parts[0])
    if len(formatted) == 1:
        return formatted[0]
    elif len(formatted) == 2:
        return f"{formatted[0]} & {formatted[1]}"
    else:
        return ", ".join(formatted[:-1]) + ", & " + formatted[-1]


def _mla_authors(names: list[str]) -> str:
    """Format authors for MLA: 'Last, First, et al.' (first author inverted, rest normal)."""
    if not names:
        return ""
    parts = names[0].strip().split()
    if len(parts) >= 2:
        first_author = f"{parts[-1]}, {' '.join(parts[:-1])}"
    else:
        first_author = names[0]
    if len(names) == 1:
        return first_author
    elif len(names) == 2:
        return f"{first_author}, and {names[1]}"
    else:
        return f"{first_author}, et al."


def _chicago_authors(names: list[str]) -> str:
    """Format authors for Chicago: same as MLA first author inversion, then normal order."""
    if not names:
        return ""
    parts = names[0].strip().split()
    if len(parts) >= 2:
        first_author = f"{parts[-1]}, {' '.join(parts[:-1])}"
    else:
        first_author = names[0]
    if len(names) == 1:
        return first_author
    elif len(names) == 2:
        return f"{first_author}, and {names[1]}"
    elif len(names) <= 10:
        middle = ", ".join(names[1:-1])
        return f"{first_author}, {middle}, and {names[-1]}"
    else:
        return f"{first_author}, et al."


def _bibtex_key(authors: list[str], year: int | None, title: str) -> str:
    """Generate a BibTeX citation key: lastnameyearfirstword."""
    last = ""
    if authors:
        parts = authors[0].strip().split()
        last = parts[-1].lower() if parts else "unknown"
    else:
        last = "unknown"
    yr = str(year) if year else "nd"
    # First significant word of title (skip articles)
    skip = {"a", "an", "the", "on", "in", "of", "for", "and", "to"}
    words = re.findall(r"[a-zA-Z]+", title.lower())
    first_word = "untitled"
    for w in words:
        if w not in skip:
            first_word = w
            break
    return f"{last}{yr}{first_word}"


def _escape_bibtex(text: str) -> str:
    """Escape special BibTeX characters."""
    # BibTeX special chars: & % $ # _ { } ~ ^
    for ch in ("&", "%", "$", "#", "_", "{", "}", "~", "^"):
        text = text.replace(ch, "\\" + ch)
    return text
```

**Step 2 — Format APA 7th:**

APA format depends on source type:

```python
def _format_apa(title, url, authors, year, journal, volume, issue, doi, publisher, source_type, access_date):
    parts = []

    # Authors
    author_str = _apa_authors(authors)
    if author_str:
        parts.append(author_str)

    # Year
    if year:
        parts.append(f"({year}).")
    else:
        parts.append("(n.d.).")

    # Title
    if source_type in ("peer_reviewed", "preprint") and journal:
        # Journal article: title not italicized, journal italicized
        parts.append(f"{title}.")
        journal_part = f"*{journal}*"
        if volume:
            journal_part += f", *{volume}*"
            if issue:
                journal_part += f"({issue})"
        parts.append(journal_part + ".")
    else:
        # Web page / other: title italicized
        parts.append(f"*{title}*.")

    # DOI or URL
    if doi:
        parts.append(f"https://doi.org/{doi}")
    else:
        parts.append(f"Retrieved {access_date}, from {url}")

    return " ".join(parts)
```

**Step 3 — Format MLA 9th:**

```python
def _format_mla(title, url, authors, year, journal, volume, issue, doi, publisher, source_type, access_date):
    parts = []

    # Authors
    author_str = _mla_authors(authors)
    if author_str:
        parts.append(author_str + ".")

    # Title
    if source_type in ("peer_reviewed", "preprint") and journal:
        # Article title in quotes, journal italicized
        parts.append(f'"{title}."')
        journal_part = f"*{journal}*"
        if volume:
            journal_part += f", vol. {volume}"
        if issue:
            journal_part += f", no. {issue}"
        if year:
            journal_part += f", {year}"
        parts.append(journal_part + ".")
    else:
        # Web page: title italicized
        parts.append(f"*{title}*.")
        if publisher:
            parts.append(f"{publisher},")
        if year:
            parts.append(f"{year}.")

    # DOI or URL
    if doi:
        parts.append(f"https://doi.org/{doi}.")
    else:
        parts.append(f"Accessed {access_date}. {url}.")

    return " ".join(parts)
```

**Step 5 — Format BibTeX:**

```python
def _format_bibtex(title, url, authors, year, journal, volume, issue, doi, publisher, source_type):
    key = _bibtex_key(authors, year, title)

    # Determine entry type
    if source_type in ("peer_reviewed",) and journal:
        entry_type = "article"
    elif source_type == "book":
        entry_type = "book"
    elif source_type == "preprint":
        entry_type = "unpublished"
    else:
        entry_type = "misc"

    safe_title = _escape_bibtex(title)

    lines = [f"@{entry_type}{{{key},"]
    lines.append(f'  title = {{{safe_title}}},')

    if authors:
        # BibTeX author format: "Last, First and Last, First"
        bib_authors = []
        for name in authors:
            parts = name.strip().split()
            if len(parts) >= 2:
                bib_authors.append(f"{parts[-1]}, {' '.join(parts[:-1])}")
            else:
                bib_authors.append(name)
        lines.append(f'  author = {{{" and ".join(bib_authors)}}},')

    if year:
        lines.append(f"  year = {{{year}}},")

    if journal:
        safe_journal = _escape_bibtex(journal)
        lines.append(f"  journal = {{{safe_journal}}},")

    if volume:
        lines.append(f"  volume = {{{volume}}},")

    if issue:
        lines.append(f"  number = {{{issue}}},")

    if publisher:
        safe_publisher = _escape_bibtex(publisher)
        lines.append(f"  publisher = {{{safe_publisher}}},")

    if doi:
        lines.append(f"  doi = {{{doi}}},")

    lines.append(f"  url = {{{url}}},")
    lines.append("}")

    return "\n".join(lines)
```

**Step 6 — Assemble the main function:**

```python
def format_citations(
    title, url, authors=None, year=None, journal=None, volume=None,
    issue=None, doi=None, publisher=None, source_type="unknown", access_date=None,
) -> dict:
    authors = authors or []
    access_date = access_date or datetime.now(timezone.utc).strftime("%Y-%m-%d")
    missing_fields = []

    if not authors:
        missing_fields.append("authors")
    if year is None:
        missing_fields.append("year")
    if source_type in ("peer_reviewed", "preprint") and not journal:
        missing_fields.append("journal")
    if not doi:
        missing_fields.append("doi")

    apa = _format_apa(title, url, authors, year, journal, volume, issue, doi, publisher, source_type, access_date)
    mla = _format_mla(title, url, authors, year, journal, volume, issue, doi, publisher, source_type, access_date)
    chicago = _format_chicago(title, url, authors, year, journal, volume, issue, doi, publisher, source_type, access_date)
    bibtex = _format_bibtex(title, url, authors, year, journal, volume, issue, doi, publisher, source_type)

    return {
        "apa": apa,
        "mla": mla,
        "chicago": chicago,
        "bibtex": bibtex,
        "missing_fields": missing_fields,
    }
```

### Formatting edge cases to handle

**No authors:** Use title as first element. APA says use title in author position. MLA says start with title. Both are standard for web pages.
```python
# In _format_apa, _format_mla, _format_chicago:
# If author_str is empty, don't append an empty string — just skip to the next part.
# The title will be the first element.
```

**No year:** Use `"(n.d.)"` in APA, omit in MLA/Chicago. In BibTeX, omit the year field entirely.

**No journal for peer-reviewed source:** Format as web resource instead. Track in `missing_fields`.

**Very long author lists (>20):** APA 7th says list first 19 authors, then ellipsis, then last author. Implement this:
```python
# In _apa_authors, after the existing logic, add:
if len(formatted) > 20:
    return ", ".join(formatted[:19]) + ", ... " + formatted[-1]
```

**DOI formatting:** Always format as full URL: `https://doi.org/{doi}`. Do NOT use `doi:` prefix.

**Title capitalization:** Do NOT change the title's capitalization. Use it as-is from the source. APA technically requires sentence case for article titles, but we don't have the information to reliably convert — better to preserve the original.

### What NOT to do

- Do NOT call any external APIs. This is pure string formatting.
- Do NOT attempt to look up missing metadata. That was Phase 8's job.
- Do NOT generate page ranges (we don't have article page ranges from Crossref consistently).
- Do NOT try to determine the "container" for every source type. Keep the logic simple: journal articles get journal formatting, everything else gets web formatting.
- Do NOT add `pages` field handling. It's not reliably available and would add complexity for minimal value.

```python
@app.get("/export/{query_id}")
async def export_citations(query_id: str, format: str = "bibtex"):
    """
    Export citations from a previous /ask query.

    Args:
        query_id: The query_id from a previous AskResponse.
        format: One of "bibtex", "json", "apa", "mla", "chicago".

    Returns:
        ExportResponse with formatted citation content.
    """
    # Validate format
    valid_formats = {"bibtex", "json", "apa", "mla", "chicago"}
    if format not in valid_formats:
        raise HTTPException(
            status_code=400,
            detail=f"Invalid format '{format}'. Must be one of: {', '.join(sorted(valid_formats))}"
        )

    # Look up cached result
    if query_id not in _query_result_cache:
        raise HTTPException(
            status_code=404,
            detail="Query result not found. Results expire after newer queries are processed. Please run the query again."
        )

    cached = _query_result_cache[query_id]
    query = cached["query"]
    sources = cached["sources"]

    # Filter to sources that were actually used (have citations)
    cited_sources = [s for s in sources if s.citations is not None]

    missing_metadata_sources = []

    if format == "bibtex":
        # Combine all BibTeX entries
        entries = []
        for s in cited_sources:
            entries.append(s.citations.bibtex)
            if s.citations.missing_fields:
                missing_metadata_sources.append(s.url)
        content = "\n\n".join(entries)

    elif format == "json":
        # Structured JSON with all formats
        import json
        export_data = []
        for s in cited_sources:
            entry = {
                "title": s.title,
                "url": s.url,
                "source_type": s.source_type,
                "citations": {
                    "apa": s.citations.apa,
                    "mla": s.citations.mla,
                    "chicago": s.citations.chicago,
                    "bibtex": s.citations.bibtex,
                },
                "missing_fields": s.citations.missing_fields,
            }
            if s.pdf_metadata:
                entry["doi"] = s.pdf_metadata.get("doi")
                entry["year"] = s.pdf_metadata.get("year")
                entry["authors"] = s.pdf_metadata.get("authors", [])
            export_data.append(entry)
            if s.citations.missing_fields:
                missing_metadata_sources.append(s.url)
        content = json.dumps(export_data, indent=2)

    else:
        # APA, MLA, or Chicago — plain text list
        lines = []
        for s in cited_sources:
            citation_text = getattr(s.citations, format, "")
            if citation_text:
                lines.append(citation_text)
            if s.citations.missing_fields:
                missing_metadata_sources.append(s.url)
        content = "\n\n".join(lines)

    return ExportResponse(
        query_id=query_id,
        query=query,
        format=format,
        sources_count=len(cited_sources),
        content=content,
        missing_metadata_sources=missing_metadata_sources,
    )
```

```python
from app.services.citation.formatter import format_citations

result = format_citations(
    title="Attention Is All You Need",
    url="https://arxiv.org/abs/1706.03762",
    authors=["Ashish Vaswani", "Noam Shazeer", "Niki Parmar", "Jakob Uszkoreit"],
    year=2017,
    journal="Advances in Neural Information Processing Systems",
    volume="30",
    issue=None,
    doi="10.48550/arXiv.1706.03762",
    publisher="NeurIPS",
    source_type="peer_reviewed",
)

print("APA:", result["apa"])
# Expected pattern: Vaswani, A., Shazeer, N., Parmar, N., & Uszkoreit, J. (2017). Attention Is All You Need. *Advances in Neural Information Processing Systems*, *30*. https://doi.org/10.48550/arXiv.1706.03762

print("\nMLA:", result["mla"])
# Expected pattern: Vaswani, Ashish, et al. "Attention Is All You Need." *Advances in Neural Information Processing Systems*, vol. 30, 2017. https://doi.org/10.48550/arXiv.1706.03762.

print("\nChicago:", result["chicago"])
# Expected pattern: Vaswani, Ashish, Noam Shazeer, Niki Parmar, and Jakob Uszkoreit. "Attention Is All You Need." *Advances in Neural Information Processing Systems* 30 (2017). https://doi.org/10.48550/arXiv.1706.03762.

print("\nBibTeX:", result["bibtex"])
# Expected pattern: @article{vaswani2017attention, title = {...}, author = {...}, ...}

print("\nMissing:", result["missing_fields"])
# Expected: [] or ["issue"] — all major fields present

assert "Vaswani" in result["apa"]
assert "2017" in result["apa"]
assert result["bibtex"].startswith("@article{")
```

## COMMON MISTAKES TO AVOID

1. **Formatting citations for only PDF sources.** Every source gets citations — HTML and PDF alike. HTML sources with no metadata get minimal web citations (title + URL + access date). The formatter handles this via the fallback paths.

2. **Crashing on None pdf_metadata.** When extracting metadata for the formatter, always do `pdf_meta = source.pdf_metadata or {}` and then `pdf_meta.get("field")`. HTML sources have `pdf_metadata=None`. If you write `source.pdf_metadata.get("authors")` without the None check, every HTML source crashes.

3. **Forgetting to generate the query_id early.** The `query_id` must be generated at the START of the `/ask` handler (or at least before the response is constructed). If you generate it inside the response constructor, it won't be available for the cache storage step.

4. **Cache growing unbounded.** The `_query_result_cache` must have a max size. Use the `OrderedDict` with LRU eviction pattern shown above. Without eviction, memory grows with every request.

5. **Storing the entire AskResponse in the cache.** Only store what the export endpoint needs: `query` and `sources`. Storing the full response (with all chunks, top_chunks, evidence_blocks) wastes memory. The export endpoint only needs source metadata and citations.

6. **BibTeX keys colliding.** Two sources by the same first author in the same year with the same first title word will get the same key. This is acceptable for Phase 11 — BibTeX tools handle duplicates. Do NOT add complex deduplication logic.

7. **Forgetting the `__init__.py` file.** Without `app/services/citation/__init__.py`, Python can't import the formatter module. You get `ModuleNotFoundError`.

8. **Changing title capitalization.** APA technically requires sentence case for article titles. Do NOT implement this — reliable case conversion requires knowing which words are proper nouns, which we can't determine. Preserve the original title as-is.

9. **Adding new dependencies.** Phase 11 needs ZERO new packages. Citation formatting is string manipulation. Do not add `citeproc`, `pybtex`, `biblib`, or any other citation library. The custom formatter is simpler, more predictable, and has no dependency chain.

10. **Making the export endpoint async when it doesn't need to be.** The export endpoint reads from an in-memory dict and formats strings. It makes no network calls. It can be async (FastAPI handles both), but make sure you're not accidentally `await`-ing synchronous operations.
